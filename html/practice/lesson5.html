          let emptyCount = 0; 
          
          for (let x = 0; x < arr.length; x++) { 
            let y=arr[x].length; 
            
            if (y > emptyCount) { 
              emptyCount = y; // здесь мы перезаписуем кол-во букв
            } 
            
            if (y == emptyCount) { 
              emptyWord = arr[x]; // здесь мы записуем это слово
