               let triangle10 = function(len) { 
                 let sp = ' '; 
                 let lt = '\'; 
                 let rt = '⁄'; 
                 let ly = '_'; 
                 
                 let lenIn = (size) => ' '.repeat(size * 4); 
                 let lenOut = (size) => ' '.repeat(size * 2); 

                 let str = lenOut(len) + rt + lt; 
                 for (let i = 0; i < len; i++) { 
                   let outK=len; 
                   let inK=0; 
                   for (let z=0; z < i; z++) { 
                     outK--; inK++; 
                     
                     if(i==len-1) { 
                      str=sp + sp + rt + ly.repeat(inK*2) + lt; 
                      continue; 
                    }; 
                    str=lenOut(outK) + rt + lenIn(inK) + lt; 
                  } 
                  div(str, result10, false, false);
                } 
              };
