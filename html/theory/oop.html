    3. Обьект с замыкающим методом 
                    
                  let Count = function () { 
                    let num = 0; 
                    return { 
                      getNext: function() { 
                        return ++num 
                      }, 
                      set: function(number) { 
                        num = number;
                      }, 
                      reset: function() { 
                        num = 0; 
                      } 
                    } 
                  } 
        
                  let counter = Count(); 
                  counter.getNext() // 1
                  counter.getNext() // 2
                  
                  counter.set(10); 
                  counter.getNext() // 11
                  counter.getNext() // 12
                  
                  counter.reset(); 
                  counter.getNext() // 1
                  counter.getNext() // 2

    2. Протопиный стиль
    Воспользовавшись SomeConstructor.prototype.method делаем все методы конструктора протопипами. А свойства будут хранится в обьекте экземпляра Animal. Кстати прототипное ООП работает быстрее функционального 
                    
                  let Animal = function() { 
                    let move = 'How its happend here?'; 
                    this.name = 'name'; 
                    this.move = move; 
                    this.speed = 0; 
                  }; 
                  
                  Animal.prototype.dang = function(dang) { 
                    switch (dang) { 
                      case 1: 
                        return 'It so sweety you can touct it ' + this.name; 
                      case 2: 
                        return 'It not dangerous but better watch out ' + this.name; 
                      case 3: 
                        return 'Give me a riffle I found something to eat for diner' + this.name; 
                        
                      default:
                        return 'Dont stay closer ' + this.name; 
                    } 
                  } 
                    
                  Animal.prototype.run = function(speed) { 
                    return this.speed += speed; 
                  } 
                  Animal.prototype.stop = function() { 
                    return this.speed = 0; 
                  } 
                  
                  let gorilla = new Animal(); 
                  gorilla.name = 'gorilla'; 
                  gorilla.run(6); 
                  gorilla.run(6);
                  gorilla.dang(); 
                  
                  console.log(gorilla.move); 
                  // Стандартная переменная move не затерлась 'How its happend here?' 
                  
                  gorilla.move = 'jump'; 
                  console.log(gorilla); 
                  //Animal { name: "gorilla", move: "jump", speed: 12 }

                  Протопипный стиль
                  let Animal = function(name) { 
                    this.name = name; //Такое переопределение в конструкторах must-have
                    // так как прокидывает ссылки на наши свойства
                  } 

                  Animal.prototype.sayHi = function () { 
                    // return name + ' said ' + this.noise; так бы оно уже не работало так как мы потеряли ссылку на свойство обьекта
                    return this.name + ' said ' + this.noise; // так все ок
                  } 
                  
                  let raven = new Animal('Raven'); 
                  raven.noise = 'kar-kar';

                  console.log(raven.sayHi()); // Raven said kar-kar
